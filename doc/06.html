我們由上一個範例修改，增加了模版模組 --template，讓程式可以重複顯示每一則送出的訊息。步驟會是這樣子：

  1. 使用者輸入文字
  2. 按下按鈕
  3. 將使用者文字送到模版
  4. 模版與使用者文字拼裝完成
  5. 顯示到到訊息區域

有註解的部分，是這次增加的地方。

--user

$(".user").vmodel("--user", true, function (){
    var vs = this;

    this.autoload = ['sned'];
    
    this.sned = function (){

        vs.root.on("click", ".send", function (){

            var text = vs.say();

            $.vmodel.get("message").show(text);

            //清空
            vs.clean();

            return false; 
        });
    }

    // 清空表單
    this.clean = function (){
        $(".user").get(0).reset();
    }

    this.say = function(){
        var val = vs.root.find(".say").val();
        return $.trim(val);
    }
});

--message

$(".message").vmodel("--message", false, function (){
    var vs = this;

    this.autoload = [];

    // 我們改由套用模版後，顯示文字
    this.show = function (val){
        
        //模版怎麼組裝，我們這裡並不關心，而是交由 --template 去處理
        var html = $.vmodel.get("template").set(val);

        // 改為插入到最前方
        vs.root.prepend(html);
    }
});

--template


// 模版模組
$(".list").vmodel("--template", true, function (){
    var vs = this;
    this.autoload = [];
    // 設定模版
    this.set = function (val){
        var html = vs.setdata(val);
        return html;
    }

    // 將資料寫入模版, 並返回 html
    this.setdata = function (val){
        var ele = vs.clone();
        ele.find(".something").text(val);
        return ele[0].outerHTML;
    }

    // 複製一份，並去除掉 hidden 讓他可以顯示
    this.clone = function (){
        return vs.root.clone().removeAttr("hidden");
    }
});
